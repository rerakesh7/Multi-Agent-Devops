### GCP/GKE Namespace Template
```yaml
# Generated by agent-001 | Requires approval before apply
apiVersion: v1
kind: Namespace
metadata:
  name: tenant-${TENANT_ID}-${ENV}
  labels:
    tenant.io/id: "${TENANT_ID}"
    tenant.io/tier: "${TIER}"          # standard|premium|enterprise
    tenant.io/environment: "${ENV}"
    pod-security.kubernetes.io/enforce: restricted
    pod-security.kubernetes.io/audit: restricted
    pod-security.kubernetes.io/warn: restricted
  annotations:
    tenant.io/owner: "${OWNER_EMAIL}"
    tenant.io/created-by: "agent-001"
    tenant.io/approved-by: "${APPROVER}"
    tenant.io/cost-center: "${COST_CENTER}"
---
apiVersion: v1
kind: ResourceQuota
metadata:
  name: tenant-quota
  namespace: tenant-${TENANT_ID}-${ENV}
spec:
  hard:
    requests.cpu: "${CPU_REQUEST:-4}"
    requests.memory: "${MEM_REQUEST:-8Gi}"
    limits.cpu: "${CPU_LIMIT:-8}"
    limits.memory: "${MEM_LIMIT:-16Gi}"
    pods: "${MAX_PODS:-50}"
    services: "10"
    persistentvolumeclaims: "10"
    secrets: "20"
    configmaps: "20"
    services.loadbalancers: "2"
    services.nodeports: "0"       # No NodePorts for tenants
---
apiVersion: v1
kind: LimitRange
metadata:
  name: tenant-limitrange
  namespace: tenant-${TENANT_ID}-${ENV}
spec:
  limits:
  - default:
      cpu: "500m"
      memory: "512Mi"
    defaultRequest:
      cpu: "100m"
      memory: "128Mi"
    type: Container
---
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: default-deny-all
  namespace: tenant-${TENANT_ID}-${ENV}
spec:
  podSelector: {}
  policyTypes:
  - Ingress
  - Egress
---
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-intra-namespace
  namespace: tenant-${TENANT_ID}-${ENV}
spec:
  podSelector: {}
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          tenant.io/id: "${TENANT_ID}"
  egress:
  - to:
    - namespaceSelector:
        matchLabels:
          tenant.io/id: "${TENANT_ID}"
  - ports:                          # Allow DNS always
    - port: 53
      protocol: UDP
    - port: 53
      protocol: TCP
---
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-monitoring
  namespace: tenant-${TENANT_ID}-${ENV}
spec:
  podSelector: {}
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          kubernetes.io/metadata.name: monitoring
    ports:
    - port: 9090    # Prometheus scrape
    - port: 8080
```

## Skill: RBAC Generation

```yaml
# RBAC for tenant team — view/deploy within own namespace only
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: tenant-developer
  namespace: tenant-${TENANT_ID}-${ENV}
rules:
- apiGroups: ["", "apps", "batch"]
  resources: ["pods", "deployments", "services", "configmaps", "jobs", "cronjobs"]
  verbs: ["get", "list", "watch", "create", "update", "patch"]
- apiGroups: [""]
  resources: ["secrets"]
  verbs: ["get", "list"]           # Read only — no secret creation
- apiGroups: [""]
  resources: ["pods/log", "pods/exec"]
  verbs: ["get", "list"]           # Exec only in non-prod
- apiGroups: ["apps"]
  resources: ["deployments/scale"]
  verbs: ["update"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: tenant-developer-binding
  namespace: tenant-${TENANT_ID}-${ENV}
subjects:
- kind: Group
  name: "tenant-${TENANT_ID}-developers"   # Maps to GCP group / AWS SSO / AAD group
  apiGroup: rbac.authorization.k8s.io
roleRef:
  kind: Role
  name: tenant-developer
  apiGroup: rbac.authorization.k8s.io
```

## Skill: Cross-Cloud Isolation Audit

```bash
#!/bin/bash
# agent-001-audit.sh — Run isolation audit across all tenants
# HITL: Results presented before any remediation

REPORT_FILE="/tmp/isolation-audit-$(date +%Y%m%d-%H%M%S).json"

echo "=== Multi-Tenant Isolation Audit ==="
echo "Timestamp: $(date -u)"
echo ""

# Check 1: Namespaces missing NetworkPolicy
echo ">> Finding namespaces with no NetworkPolicy..."
for ns in $(kubectl get ns -l tenant.io/id -o jsonpath='{.items[*].metadata.name}'); do
  count=$(kubectl get networkpolicy -n "$ns" --no-headers 2>/dev/null | wc -l)
  if [ "$count" -eq 0 ]; then
    echo "  ❌ VIOLATION: $ns has NO NetworkPolicy"
  else
    echo "  ✅ $ns has $count NetworkPolicy/ies"
  fi
done

# Check 2: ResourceQuota missing
echo ""
echo ">> Checking ResourceQuotas..."
for ns in $(kubectl get ns -l tenant.io/id -o jsonpath='{.items[*].metadata.name}'); do
  if ! kubectl get resourcequota -n "$ns" &>/dev/null; then
    echo "  ❌ VIOLATION: $ns missing ResourceQuota"
  fi
done

# Check 3: Pods without resource limits
echo ""
echo ">> Pods without resource limits..."
kubectl get pods --all-namespaces -l tenant.io/id \
  -o jsonpath='{range .items[*]}{.metadata.namespace}/{.metadata.name}: {.spec.containers[*].resources.limits}{"\n"}{end}' \
  | grep -v "cpu\|memory" || echo "  ✅ All pods have resource limits"

# Check 4: Cross-namespace service account usage
echo ""
echo ">> Checking for cross-namespace ClusterRoleBindings..."
kubectl get clusterrolebindings -o json | jq -r '
  .items[] | 
  select(.subjects[]?.namespace != null) |
  "  ⚠️  ClusterRoleBinding \(.metadata.name) references namespace \(.subjects[].namespace)"
'

echo ""
echo "=== Audit Complete. Review before applying any fixes. ==="
echo "[HUMAN REVIEW REQUIRED before remediation]"
```

## Runbook: Tenant Breach Response

```
INCIDENT: Cross-tenant data access detected
SEVERITY: P0 — CRITICAL

STEPS (all require human approval before execution):

1. ISOLATE
   □ Confirm breach scope (which tenants affected)
   □ [APPROVE] Apply emergency NetworkPolicy deny-all to affected namespace
   □ [APPROVE] Rotate all secrets in affected namespace
   
2. INVESTIGATE
   □ Pull audit logs: kubectl get events -n <ns> --sort-by='.lastTimestamp'
   □ Pull API server audit logs from cloud provider
   □ Identify compromised service account

3. REMEDIATE  
   □ [APPROVE] Revoke compromised SA tokens
   □ [APPROVE] Apply missing NetworkPolicies
   □ [APPROVE] Update RBAC to principle of least privilege

4. VERIFY
   □ Re-run isolation audit
   □ Confirm no cross-namespace traffic
   □ Update runbook with new attack vector
```
